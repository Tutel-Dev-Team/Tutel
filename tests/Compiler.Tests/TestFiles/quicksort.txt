func void quickSort(int[] arr, int low, int high) {
    if (low < high) {
        // Разделение массива
        int pi = partition(arr, low, high);
        
        // Рекурсивная сортировка левой и правой частей
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

func int partition(int[] arr, int low, int high) {
    int pivot = arr[high];
    int i = low - 1; 
    
    for (int j = low; j < high; j = j + 1) {
        // Если текущий элемент меньше или равен опорному
        if (arr[j] <= pivot) {
            i = i + 1;
            
            // Меняем местами arr[i] и arr[j]
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
    
    int temp = arr[i + 1];
    arr[i + 1] = arr[high];
    arr[high] = temp;
    
    return i + 1;
}

func int main(){
    int n = 10000;
    int[] arr = new int[n];
    for (int i = 0; i < n; i = i + 1){
	    arr[i] = n - i;
    }
    quickSort(arr, 0, len(arr) - 1);
    return 0;
}
