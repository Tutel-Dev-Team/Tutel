// Решето Эратосфена - находит все простые числа до N
func int[] sieveOfEratosthenes(int n) {
    // Если n < 2, возвращаем пустой массив
    if (n < 2) {
        return new int[0];
    }
    
    // Создаем массив для отметки простых чисел (0 - не простое, 1 - простое)
    int[] isPrime = new int[n + 1];
    
    // Изначально считаем все числа простыми (кроме 0 и 1)
    for (int i = 0; i < n + 1; i = i + 1) {
        isPrime[i] = 1;
    }
    isPrime[0] = 0;
    isPrime[1] = 0;
    
    // Основной алгоритм
    int p = 2;
    while (p * p <= n) {
        // Если p простое
        if (isPrime[p] == 1) {
            // Помечаем все кратные p как составные
            for (int i = p * p; i <= n; i = i + p) {
                isPrime[i] = 0;
            }
        }
        p = p + 1;
    }
    
    // Считаем количество простых чисел
    int count = 0;
    for (i = 2; i <= n; i = i + 1) {
        if (isPrime[i] == 1) {
            count = count + 1;
        }
    }
    
    // Создаем массив с простыми числами
    int[] primes = new int[count];
    int index = 0;
    for (i = 2; i <= n; i = i + 1) {
        if (isPrime[i] == 1) {
            primes[index] = i;
            index = index + 1;
        }
    }
    
    return primes;
}

func int main(){
    int[] primes = sieveOfEratosthenes(1000000);
    return 0;
}
